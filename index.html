function calculateRank() { const score = parseFloat(document.getElementById("score").value); const avg = parseFloat(document.getElementById("avg").value); const stdDev = parseFloat(document.getElementById("stdDev").value); const totalStudents = parseInt(document.getElementById("totalStudents").value); // z-score 계산 const zScore = (score - avg) / stdDev; const percentileRank = 0.5 * (1 + erf(zScore / Math.sqrt(2))) * 100; // 석차 계산 const rank = Math.round(totalStudents * (1 - percentileRank / 100)); // 상위 퍼센트 계산 const topPercent = ((totalStudents - rank + 1) / totalStudents) * 100; // 결과 표시 document.getElementById("result").innerHTML = `전체 ${totalStudents}명 중 ${rank}등이며, 상위 ${topPercent.toFixed(2)}%입니다.`; } // Error function approximation function erf(x) { const a1 = 0.254829592; const a2 = -0.284496736; const a3 = 1.421413741; const a4 = -1.453152027; const a5 = 1.061405429; const p = 0.3275911; const sign = x < 0 ? -1 : 1; x = Math.abs(x); const t = 1 / (1 + p * x); const y = 1 - (((((a5 * t + a4) * t) + a3) * t + a2) * t + a1) * t * Math.exp(-x * x); return sign * y; }
